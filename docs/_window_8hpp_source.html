210  template<EventSupport E>
211  Event buildEvent(const SDL_Event sdlEvent) const
212  {
213  try
214  {
215  if constexpr(E.mouseMotionEventsEnabled)
216  {
217  if(sdlEvent.type == SDL_MOUSEMOTION)
218  return Event(std::make_unique<MouseMotionEvent>(MouseMotionEvent(sdlEvent.motion)));
219  }
220 
221  if constexpr(E.mouseButtonEventsEnabled)
222  {
223  if(sdlEvent.type == SDL_MOUSEBUTTONDOWN || sdlEvent.type == SDL_MOUSEBUTTONUP)
224  return Event(std::make_unique<MouseButtonEvent>(MouseButtonEvent(sdlEvent.button)));
225  }
226 
227  if constexpr(E.keyboardEventsEnabled)
228  {
229  if(sdlEvent.type == SDL_KEYDOWN || sdlEvent.type == SDL_KEYUP)
230  return Event(std::make_unique<KeyboardEvent>(KeyboardEvent(sdlEvent.key)));
231  }
232 
233  if constexpr(E.mouseWheelEventsEnabled)
234  {
235  if(sdlEvent.type == SDL_MOUSEWHEEL)
236  return Event(std::make_unique<MouseWheelEvent>(MouseWheelEvent(sdlEvent.wheel)));
237  }
238 
239  if constexpr(E.windowEventsEnabled)
240  {
241  if(sdlEvent.type == SDL_WINDOWEVENT)
242  return Event(std::make_unique<WindowEvent>(WindowEvent(sdlEvent.window)));
243  }
244 
245  if(sdlEvent.type == SDL_QUIT)
246  return Event(std::make_unique<QuitEvent>());
247 
248  return Event(nullptr);
249  }
250  catch(const SdlException&)
251  {
252  return Event(nullptr);
253  }
254  }
